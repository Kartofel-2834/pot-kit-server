// Types
import type { Ref } from 'vue';

export const <%upper%>_ATTACHED_BOX_POSITION = {
    TOP_START: 'top-start',
    TOP_END: 'top-end',
    TOP_CENTER: 'top-center',

    BOTTOM_START: 'bottom-start',
    BOTTOM_END: 'bottom-end',
    BOTTOM_CENTER: 'bottom-center',

    LEFT_START: 'left-start',
    LEFT_END: 'left-end',
    LEFT_CENTER: 'left-center',

    RIGHT_START: 'right-start',
    RIGHT_END: 'right-end',
    RIGHT_CENTER: 'right-center',
} as const;

export type E<%camel%>AttachedBoxPosition =
    (typeof <%upper%>_ATTACHED_BOX_POSITION)[keyof typeof <%upper%>_ATTACHED_BOX_POSITION];

export interface I<%camel%>AttachTargetProps {
    /** Box element */
    box: Element | null;

    /** Box position that relative to target */
    position?: E<%camel%>AttachedBoxPosition;

    /** Distance between box and target */
    nudge?: number;

    /** Distance between box and edge */
    edgeMargin?: number;

    /** Fixed box position without moving to opposite sides */
    persistent?: boolean;

    /** Sticky box that will follow target */
    sticky?: boolean;

    /** Target element */
    target?: Element | null;

    /** The side to which the tooltip will be moved if it does not fit in the X direction */
    oppositeSideX?: E<%camel%>AttachedBoxPosition;

    /** The side to which the tooltip will be moved if it does not fit in the Y direction */
    oppositeSideY?: E<%camel%>AttachedBoxPosition;
}

export interface I<%camel%>AttachTargetExpose {
    /** Box top-left corner coordinates on screen */
    boxCoordinates: Readonly<Ref<[x: number, y: number]>>;

    /** Target element */
    target: Readonly<Ref<Element | null>>;
}
