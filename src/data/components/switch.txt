<script lang="ts" generic="TRUE_VALUE = true, FALSE_VALUE = false" setup>
// Types
import type { I<%camel%>SwitchEmits, I<%camel%>SwitchExpose, I<%camel%>SwitchProps } from '<%typesImport%>/components/switch';

// Vue
import { computed, toRef, useTemplateRef } from 'vue';

// Composables
import { useClassList } from '<%composablesImport%>/class-list';
import { useDeviceProperties } from '<%composablesImport%>/device-is';

const props = withDefaults(defineProps<I<%camel%>SwitchProps<TRUE_VALUE, FALSE_VALUE>>(), {
    value: undefined,
    modelValue: undefined,
    trueValue: undefined,
    falseValue: undefined,
    inputName: 'switch',
    transition: '<%kebab%>-switch-transition',
});

const emit = defineEmits<I<%camel%>SwitchEmits<TRUE_VALUE, FALSE_VALUE>>();

// Data
const input = useTemplateRef('input');

// Computed
const currentValue = computed(() => {
    return props.value ?? props.modelValue ?? false;
});

const trueValue = computed(
    () => (props.trueValue === undefined ? true : props.trueValue) as TRUE_VALUE,
);

const falseValue = computed(
    () => (props.falseValue === undefined ? false : props.falseValue) as FALSE_VALUE,
);

const isChecked = computed(() => currentValue.value === trueValue.value);

// Composables
const $properties = useDeviceProperties(
    {
        size: toRef(() => props.size),
        color: toRef(() => props.color),
        radius: toRef(() => props.radius),
    },
    toRef(() => props.devices),
);

const $classList = useClassList(
    {
        size: $properties.size,
        color: $properties.color,
        radius: $properties.radius,
        checked: isChecked,
        disabled: toRef(() => props.disabled),
        invalid: toRef(() => props.invalid),
    },
    'switch',
);

// Methods
function onChange() {
    const newValue = isChecked.value ? falseValue.value : trueValue.value;
    emit('update:modelValue', newValue);
    emit('change', newValue);
}

// Exports
defineExpose<Readonly<I<%camel%>SwitchExpose>>({
    input: input,
});
</script>

<template>
    <label
        :class="['<%kebab%>-switch', $classList]"
        @mousedown.prevent
    >
        <input
            ref="input"
            type="checkbox"
            class="<%kebab%>-switch-input"
            :checked="isChecked"
            :name="inputName"
            :value="currentValue"
            :disabled="disabled"
            @change.stop="onChange"
        />

        <span class="<%kebab%>-switch-track">
            <span class="<%kebab%>-switch-thumb" />
        </span>

        <span
            v-if="$slots.default"
            class="<%kebab%>-switch-content"
        >
            <slot />
        </span>
    </label>
</template>
