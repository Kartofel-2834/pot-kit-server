<script lang="ts" setup>
// Types
import type { I%camel%ButtonProps } from '$types-import/components/button';
import type { E%camel%Device, E%camel%Color, E%camel%Size, E%camel%Radius } from '$types-import/index';

// Enums
import { %upper%_BREAKPOINT } from '$types-import/index';

// Vue
import { defineAsyncComponent, computed } from 'vue';

// Composables
import { useClassList } from '$composables-import/class-list';
import { useDeviceProperties } from '$composables-import/device-properties';
import { useDeviceIs } from '$composables-import/device-is';

// Components
const %camel%Icon = defineAsyncComponent(() => import('./%camel%Icon.vue'));

const $props = withDefaults(
    defineProps<I%camel%ButtonProps<E%camel%Device, E%camel%Color, E%camel%Size, E%camel%Radius>>(), 
    {
        tag: 'button',
    },
);

const $deviceIs = useDeviceIs(%upper%_BREAKPOINT);

/**
 * Вычисляет и возвращает свойства компонента на основе
 * брейкпоинтов и текущего размера экрана
 */
const properties = computed(() => {
    return useDeviceProperties(
        {
            color: $props.color,
            size: $props.size,
            radius: $props.radius,
        },
        $props.devices,
        $deviceIs.device.value,
        %upper%_BREAKPOINT,
    );
});

/** Классы модификаторы компонента */
const classList = computed(() =>
    useClassList({
        ...properties.value,
        square: $props.square,
    }),
);
</script>

<template>
    <component
        :is="tag"
        :class="['%kebab%-button', classList]"
        :disabled="disabled"
    >
        <slot name="preicon">
            <%camel%Icon
                v-if="preicon"
                :icon="preicon"
                class="%kebab%-button__icon %kebab%-button__icon_pre"
            />
        </slot>

        <span
            v-if="$slots?.default"
            class="%kebab%-button__label"
        >
            <slot />
        </span>

        <slot name="icon">
            <%camel%Icon
                v-if="icon"
                :icon="icon"
                class="%kebab%-button__icon %kebab%-button__icon_post"
            />
        </slot>
    </component>
</template>

<style>
.%kebab%-button {
    display: flex;
    align-items: center;
    justify-content: center;
    outline: none;
    cursor: pointer;
    user-select: none;
    border-style: solid;
    font-weight: 500;
    line-height: 1;
    transition:
        color 0.4s ease,
        border-color 0.4s ease,
        background-color 0.4s ease;
}

/* --- %camel%Button - Disabled --- */
.%kebab%-button:disabled {
    cursor: not-allowed;
}

/* --- %camel%Button - Square --- */
.%kebab%-button._square {
    padding: 0;
    aspect-ratio: 1 / 1;
}

.%kebab%-button__icon {
    flex-shrink: 0;
}
</style>
