<script lang="ts" setup>
// Types
import type { I<%camel%>AccordionProps, I<%camel%>AccordionSlots } from '<%typesImport%>/components/accordion';

// Vue
import { ref, computed, watch, onUnmounted, onMounted } from 'vue';

// Composables
import { useClassList } from '<%composablesImport%>/class-list';
import { useThrottle } from '<%composablesImport%>/timer';

const $slots = defineSlots<I<%camel%>AccordionSlots>();

const $props = withDefaults(defineProps<I<%camel%>AccordionProps>(), {
    opened: undefined,
    modelValue: undefined,
    disabled: false,
});

const $emit = defineEmits<{
    open: [];
    close: [];
    'update:modelValue': [isVisible: boolean];
}>();

// Data
const content = ref<Element | null>(null);
const contentHeight = ref<number>(NaN);

const contentResizeObserver = new ResizeObserver(
    useThrottle({
        delay: 500,
        action: updateContentHeight,
    }),
);

// Lifecycle
onMounted(() => setTimeout(updateContentHeight));
onUnmounted(() => contentResizeObserver.disconnect());

// Computed
const isOpen = computed(() => Boolean(!$props.disabled && ($props.opened ?? $props.modelValue)));

const classList = computed(() =>
    useClassList({
        opened: isOpen.value,
        disabled: $props.disabled,
    }),
);

const currentStyles = computed(() => {
    return {
        '--<%kebab%>-accordion-content-height': `${contentHeight.value || 1000}px`,
    };
});

// Watchers
watch(
    () => content.value,
    (newContent, oldContent) => {
        if (newContent) contentResizeObserver.observe(newContent);
        if (oldContent) contentResizeObserver.unobserve(oldContent);
    },
);

// Methods
function toggle() {
    if ($props.disabled) return;

    if (isOpen.value) {
        close();
    } else {
        open();
    }
}

function open() {
    if ($props.disabled) return;

    $emit('open');
    $emit('update:modelValue', true);
}

function close() {
    if ($props.disabled) return;

    $emit('close');
    $emit('update:modelValue', false);
}

function updateContentHeight() {
    if (
        content.value &&
        content.value.scrollHeight &&
        content.value.scrollHeight !== contentHeight.value
    ) {
        contentHeight.value = content.value.scrollHeight;
    }
}
</script>

<template>
    <div
        :class="['<%kebab%>-accordion', classList]"
        :style="currentStyles"
    >
        <slot
            name="header"
            :toggle="toggle"
            :open="open"
            :close="close"
        >
            <div
                v-if="$slots.title"
                data-<%kebab%>-accordion-header
                class="<%kebab%>-accordion__header"
                :tabindex="disabled ? -1 : 0"
                @click="toggle"
                @keydown.enter="toggle"
                @keydown.space="toggle"
            >
                <div class="<%kebab%>-accordion__header__title">
                    <slot name="title" />
                </div>

                <div class="<%kebab%>-accordion__header__icon">
                    <slot name="icon">
                        <svg
                            xmlns="http://www.w3.org/2000/svg"
                            viewBox="0 -960 960 960"
                            width="24px"
                            fill="currentColor"
                        >
                            <path d="M480-344 240-584l56-56 184 184 184-184 56 56-240 240Z" />
                        </svg>
                    </slot>
                </div>
            </div>
        </slot>

        <Transition name="<%kebab%>-accordion-transition">
            <div
                v-show="isOpen"
                class="<%kebab%>-accordion__content"
                ref="content"
            >
                <slot />
            </div>
        </Transition>
    </div>
</template>
