<script lang="ts">
// Types
import type { I<%camel%>SlotTargetProps } from '<%typesImport%>/components/slot-target';

// Vue
import { cloneVNode, defineComponent } from 'vue';

export default defineComponent<I<%camel%>SlotTargetProps>({
    setup(props, { slots }) {
        return () => {
            if (!slots?.default) {
                return null;
            }

            const vNodesList = slots.default();

            return Array.isArray(vNodesList) && vNodesList.length && props.options
                ? vNodesList.map(vNode => cloneVNode(vNode, props.options))
                : vNodesList;
        };
    },
});
</script>
